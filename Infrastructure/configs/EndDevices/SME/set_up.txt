# === INSTALACIÓN DE DEPENDENCIAS DEL SISTEMA (ALPINE LINUX) ===
apk update && apk add bash curl nano iputils openssh-client python3 py3-pip \
    gcc musl-dev python3-dev libffi-dev openssl-dev build-base sqlite git

# === CONFIGURACIÓN DE COMPATIBILIDAD SSH (PARA CISCO IOS ANTIGUO) ===

mkdir -p ~/.ssh
echo "Host *
    KexAlgorithms +diffie-hellman-group1-sha1
    HostKeyAlgorithms +ssh-rsa
    Ciphers +aes128-cbc,3des-cbc" > ~/.ssh/config
chmod 600 ~/.ssh/config


# === CLONAR ÚNICAMENTE LA CARPETA 'SME' EN EL HOME ===
# 1. Crea una carpeta vacía para tu proyecto y entra en ella
mkdir mi_proyecto_sme
cd mi_proyecto_sme

# 2. Inicializa un repositorio de Git local
git init

# 3. Conéctate a tu repositorio remoto de GitHub
git remote add origin https://github.com/EmilianoTL/ASR_FinalProject.git

# 4. Activa la característica de descarga dispersa indicando la carpeta exacta
git sparse-checkout set SME

# 5. Descarga (Pull) exclusivamente la rama específica que necesitas
git pull origin feature/snmp-telemetry

# === CONFIGURACIÓN DEL ENTORNO DE PYTHON ===

python3 -m venv venv
source venv/bin/activate

pip install --upgrade pip
pip install -r requirements.txt
