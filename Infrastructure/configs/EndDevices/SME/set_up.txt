# === INSTALACIÓN DE DEPENDENCIAS DEL SISTEMA (ALPINE LINUX) ===
apk update && apk add bash curl nano iputils openssh-client python3 py3-pip \
    gcc musl-dev python3-dev libffi-dev openssl-dev build-base sqlite git

# === CONFIGURACIÓN DEL ENTORNO DE PYTHON ===

python3 -m venv venv
source venv/bin/activate

pip install --upgrade pip
pip install -r requirements.txt

# === CONFIGURACIÓN DE COMPATIBILIDAD SSH (PARA CISCO IOS ANTIGUO) ===

mkdir -p ~/.ssh
echo "Host *
    KexAlgorithms +diffie-hellman-group1-sha1
    HostKeyAlgorithms +ssh-rsa
    Ciphers +aes128-cbc,3des-cbc" > ~/.ssh/config
chmod 600 ~/.ssh/config


# === CLONAR ÚNICAMENTE LA CARPETA 'SME' EN EL HOME ===
cd home

git clone --filter=blob:none --sparse https://github.com/EmilianoTL/ASR_FinalProject.git
cd * 
REPO_DIR=$(basename "$PWD")
git sparse-checkout set SME

mv SME ../
cd ..

rm -rf "$REPO_DIR"
